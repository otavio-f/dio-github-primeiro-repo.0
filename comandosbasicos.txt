# 1: Iniciar o ssh-agent:
$ eval $(ssh-agent -s)

# Vai ter uma resposta do tipo
> Agent pid <um numero aleatorio>

# 2: Adicionar a chave privada ao ssh-agent
$ ssh-add ~/.ssh/id_ed25519
